사용 예시

<!-- 
모달의 독립성을 유지하기 위해 ModalManager 클래스에서 분리된 이벤트 처리 방식을 채택하여,
각 모달이 로드되는 시점에 페이지 또는 별도의 스크립트에서 해당 로직을 관리합니다.
-->
<script>
  document.addEventListener('DOMContentLoaded', function() {
    const previewModal = document.getElementById('testModal');
    previewModal.querySelector('.is-false').addEventListener('click', function() {
      console.log('취소 로직 실행');
      // 취소 로직 구현
      previewModal.classList.remove('is-show');
    });
    previewModal.querySelector('.is-true').addEventListener('click', function() {
      console.log('저장 로직 실행');
      // 저장 로직 구현
      previewModal.classList.remove('is-show');
    });
  });
</script>

customAlert.confirm(
  title: string,              // confirm 타이틀
  message: string,            // confirm 메시지
  onConfirm: function,        // 확인 버튼 클릭 시 실행 함수
  onCancel: function,         // 취소 버튼 클릭 시 실행 함수
  confirmButtonText: string,  // 확인 버튼 텍스트
  cancelButtonText: string    // 취소 버튼 텍스트
);
customAlert.confirm('로그아웃', '로그아웃 하시겠습니까?', logout, cancelLogout, '로그아웃', '취소');

}

  console.log('로그아웃을 취소합니다.');
  // 취소 로직 실행
}
